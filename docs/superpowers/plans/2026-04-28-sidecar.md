# Multiharness Sidecar (Plan 1 of 3) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `multiharness-sidecar` — a Bun-based WebSocket server that hosts pi-agent-core agents per workspace, exposes a JSON-RPC-ish control API on a Unix domain socket, and streams events back to clients. Compiles to a single binary the Mac app will bundle in `Multiharness.app/Contents/Resources/`.

**Architecture:** Single TypeScript codebase. WebSocket frames carry JSON-RPC requests/responses and server-pushed events. Agents are owned by an `AgentRegistry` keyed by workspace ID. Each `AgentSession` wraps one `@mariozechner/pi-agent-core` Agent plus tool implementations scoped to the workspace's worktree path. Configuration (socket path, data dir) comes from environment variables set by the Mac app's `SidecarManager`.

**Tech Stack:** Bun, TypeScript, `@mariozechner/pi-ai`, `@mariozechner/pi-agent-core`, vitest.

**Reference spec:** `docs/superpowers/specs/2026-04-28-multiharness-phase-1-design.md`

export type RpcRequest = {
  kind: "request";
  id: string;
  method: string;
  params: Record<string, unknown>;
};

export type ParsedFrame = RpcRequest;

export function parseFrame(raw: string): ParsedFrame {
  let obj: unknown;
  try {
    obj = JSON.parse(raw);
  } catch (e) {
    throw new Error(`parse error: ${(e as Error).message}`);
  }
  if (!obj || typeof obj !== "object") throw new Error("frame must be an object");
  const o = obj as Record<string, unknown>;
  if (typeof o.id !== "string") throw new Error("frame missing string id");
  if (typeof o.method !== "string") throw new Error("frame missing string method");
  const params = o.params && typeof o.params === "object" ? (o.params as Record<string, unknown>) : {};
  return { kind: "request", id: o.id, method: o.method, params };
}

export function formatResponse(id: string, result: unknown): string {
  return JSON.stringify({ id, result });
}

export function formatErrorResponse(id: string, code: string, message: string): string {
  return JSON.stringify({ id, error: { code, message } });
}

export function formatEvent(event: string, params: Record<string, unknown>): string {
  return JSON.stringify({ event, params });
}
```

import { resolve, sep } from "node:path";

export function resolveInside(worktreePath: string, candidate: string): string {
  const root = resolve(worktreePath);
  const full = resolve(root, candidate);
  if (full !== root && !full.startsWith(root + sep)) {
    throw new Error(`path is outside worktree: ${candidate}`);
  }
  return full;
}
```

import { mkdir, appendFile } from "node:fs/promises";
import { dirname } from "node:path";

export class JsonlWriter {
  private prepared = false;
  private queue: Promise<void> = Promise.resolve();

  constructor(private readonly filePath: string) {}

  async append(record: unknown): Promise<void> {
    const line = JSON.stringify(record) + "\n";
    this.queue = this.queue.then(async () => {
      if (!this.prepared) {
        await mkdir(dirname(this.filePath), { recursive: true });
        this.prepared = true;
      }
      await appendFile(this.filePath, line, "utf8");
    });
    return this.queue;
  }

  async close(): Promise<void> {
    await this.queue;
  }
}
```

type Level = "debug" | "info" | "warn" | "error";

const LEVELS: Record<Level, number> = { debug: 10, info: 20, warn: 30, error: 40 };
const threshold = LEVELS[(process.env.MULTIHARNESS_LOG_LEVEL as Level) ?? "info"] ?? LEVELS.info;

function emit(level: Level, msg: string, extra?: Record<string, unknown>): void {
  if (LEVELS[level] < threshold) return;
  const ts = new Date().toISOString();
  const payload = extra ? ` ${JSON.stringify(extra)}` : "";
  console.error(`${ts} ${level} ${msg}${payload}`);
}

export const log = {
  debug: (m: string, e?: Record<string, unknown>) => emit("debug", m, e),
  info: (m: string, e?: Record<string, unknown>) => emit("info", m, e),
  warn: (m: string, e?: Record<string, unknown>) => emit("warn", m, e),
  error: (m: string, e?: Record<string, unknown>) => emit("error", m, e),
};
```

import { log } from "./logger.js";

export function startParentPidWatchdog(intervalMs = 1000): void {
  const initial = process.ppid;
  setInterval(() => {
    const current = process.ppid;
    if (current === 1 || current !== initial) {
      log.warn("parent pid changed; exiting", { initial, current });
      process.exit(0);
    }
  }, intervalMs).unref();
}

```ts
import { readFile } from "node:fs/promises";
import { Type, type AgentTool } from "@mariozechner/pi-agent-core";
import { resolveInside } from "../pathGuard.js";
import { resolve, relative } from "node:path";

const MAX_FILE_BYTES = 5 * 1024 * 1024;

export function grepTool(worktreePath: string): AgentTool {
  return {
    name: "grep",
    description: "Search for a regular expression across files in the worktree.",
    parameters: Type.Object({
      pattern: Type.String(),
      path: Type.Optional(Type.String({ description: "Optional file or directory to search; defaults to the worktree root." })),
      glob: Type.Optional(Type.String({ description: "Optional glob filter, e.g. '**/*.ts'." })),
    }),
    execute: async ({ pattern, path, glob }: { pattern: string; path?: string; glob?: string }) => {
      const root = resolve(worktreePath);
      const re = new RegExp(pattern);
      const matches: { path: string; line: number; text: string }[] = [];
      const filter = glob ? new Bun.Glob(glob) : null;
      const startRel = path ?? ".";
      const start = resolveInside(root, startRel);

      const walked = await walk(start, root, filter);
      for (const file of walked) {
        try {
          const stat = await Bun.file(file).size;
          if (stat > MAX_FILE_BYTES) continue;
          const text = await readFile(file, "utf8");
          const lines = text.split("\n");
          for (let i = 0; i < lines.length; i++) {
            const line = lines[i] ?? "";
            if (re.test(line)) {
              matches.push({ path: relative(root, file), line: i + 1, text: line });
            }
          }
        } catch {
          // unreadable / binary; skip
        }
      }
      return { matches };
    },
  };
}

async function walk(start: string, root: string, filter: Bun.Glob | null): Promise<string[]> {
  const out: string[] = [];
  const stat = await Bun.file(start).exists();
  if (!stat) return out;
  const info = Bun.file(start);
  if ((await info.stat()).isFile()) {
    if (!filter || filter.match(relative(root, start))) out.push(start);
    return out;
  }
  // directory walk via Bun.Glob
  const g = filter ?? new Bun.Glob("**/*");
  for await (const m of g.scan({ cwd: start, onlyFiles: true })) {
    out.push(resolve(start, m));
  }
  return out;
}
```

import { Type, type AgentTool } from "@mariozechner/pi-agent-core";
import { resolveInside } from "../pathGuard.js";

const DEFAULT_TIMEOUT_MS = 120_000;

export function bashTool(worktreePath: string): AgentTool {
  return {
    name: "bash",
    description: "Run a shell command in /bin/zsh inside the workspace's worktree.",
    parameters: Type.Object({
      command: Type.String(),
      working_dir: Type.Optional(Type.String({ description: "Optional path inside worktree to use as cwd." })),
      timeout_ms: Type.Optional(Type.Number({ description: "Default 120000." })),
    }),
    execute: async ({ command, working_dir, timeout_ms }: { command: string; working_dir?: string; timeout_ms?: number }) => {
      const cwd = working_dir ? resolveInside(worktreePath, working_dir) : worktreePath;
      const proc = Bun.spawn(["/bin/zsh", "-c", command], {
        cwd,
        stdout: "pipe",
        stderr: "pipe",
        env: { ...process.env },
      });
      const timeout = timeout_ms ?? DEFAULT_TIMEOUT_MS;
      let timedOut = false;
      const timer = setTimeout(() => {
        timedOut = true;
        proc.kill();
      }, timeout);

      const [stdout, stderr, exitCode] = await Promise.all([
        new Response(proc.stdout).text(),
        new Response(proc.stderr).text(),
        proc.exited,
      ]);
      clearTimeout(timer);
      return { exitCode, stdout, stderr, timedOut };
    },
  };
}

- [ ] **Step 1: Implement `sidecar/src/agentSession.ts`**

```ts
import { Agent } from "@mariozechner/pi-agent-core";
import type { ProviderConfig } from "./providers.js";
import { buildModel } from "./providers.js";
import { buildTools } from "./tools/index.js";
import { JsonlWriter } from "./jsonl.js";
import { log } from "./logger.js";

export type SessionEvent = { type: string; [k: string]: unknown };
export type EventSink = (workspaceId: string, ev: SessionEvent) => void;

export type AgentSessionOptions = {
  workspaceId: string;
  worktreePath: string;
  systemPrompt: string;
  providerConfig: ProviderConfig;
  jsonlPath: string;
  sink: EventSink;
};

export class AgentSession {
  private agent: Agent;
  private writer: JsonlWriter;
  private seq = 0;

  constructor(private readonly opts: AgentSessionOptions) {
    this.agent = new Agent({
      initialState: {
        systemPrompt: opts.systemPrompt,
        model: buildModel(opts.providerConfig),
        tools: buildTools(opts.worktreePath),
      },
    });
    this.writer = new JsonlWriter(opts.jsonlPath);
    this.agent.subscribe((event) => this.handle(event));
  }

  async prompt(message: string): Promise<void> {
    await this.agent.prompt(message);
  }

  async continueRun(): Promise<void> {
    await this.agent.continue();
  }

  abort(): void {
    this.agent.abort();
  }

  async dispose(): Promise<void> {
    try { this.agent.abort(); } catch {}
    await this.writer.close();
  }

  private handle(event: unknown): void {
    const e = event as SessionEvent;
    this.opts.sink(this.opts.workspaceId, e);
    // Persist meaningful events (skip pure deltas to keep JSONL compact)
    const persist = e.type === "message_end" || e.type === "tool_execution_end" || e.type === "turn_end" || e.type === "agent_end" || e.type === "agent_start";
    if (persist) {
      this.writer.append({ seq: this.seq++, ts: Date.now(), event: e }).catch((err) => {
        log.warn("jsonl append failed", { err: String(err) });
      });
    }
  }
}

import { join } from "node:path";
import { AgentSession, type EventSink } from "./agentSession.js";
import type { ProviderConfig } from "./providers.js";

export type CreateOptions = {
  workspaceId: string;
  worktreePath: string;
  systemPrompt: string;
  providerConfig: ProviderConfig;
};

export class AgentRegistry {
  private sessions = new Map<string, AgentSession>();

  constructor(private readonly dataDir: string, private readonly sink: EventSink) {}

  async create(opts: CreateOptions): Promise<void> {
    if (this.sessions.has(opts.workspaceId)) {
      throw new Error(`session for workspace ${opts.workspaceId} already exists`);
    }
    const jsonlPath = join(this.dataDir, "workspaces", opts.workspaceId, "messages.jsonl");
    const session = new AgentSession({ ...opts, jsonlPath, sink: this.sink });
    this.sessions.set(opts.workspaceId, session);
  }

  get(workspaceId: string): AgentSession {
    const s = this.sessions.get(workspaceId);
    if (!s) throw new Error(`no session for workspace ${workspaceId}`);
    return s;
  }

  list(): string[] {
    return [...this.sessions.keys()];
  }

  async dispose(workspaceId: string): Promise<void> {
    const s = this.sessions.get(workspaceId);
    if (!s) return;
    await s.dispose();
    this.sessions.delete(workspaceId);
  }

  async disposeAll(): Promise<void> {
    for (const id of [...this.sessions.keys()]) await this.dispose(id);
  }
}

import { formatErrorResponse, formatResponse } from "./rpc.js";

export type Handler = (params: Record<string, unknown>) => Promise<unknown> | unknown;

export class Dispatcher {
  private handlers = new Map<string, Handler>();

  register(method: string, handler: Handler): void {
    this.handlers.set(method, handler);
  }

  async dispatch(id: string, method: string, params: Record<string, unknown>): Promise<string> {
    const h = this.handlers.get(method);
    if (!h) return formatErrorResponse(id, "METHOD_NOT_FOUND", `unknown method: ${method}`);
    try {
      const result = await h(params);
      return formatResponse(id, result ?? null);
    } catch (e) {
      const err = e as Error;
      return formatErrorResponse(id, "HANDLER_ERROR", err.message ?? String(err));
    }
  }
}

  d.register("health.ping", () => ({ pong: true, version: VERSION }));

  d.register("agent.create", async (p) => {
    const { workspaceId, worktreePath, systemPrompt, providerConfig } = p as {
      workspaceId: string;
      worktreePath: string;
      systemPrompt: string;
      providerConfig: ProviderConfig;
    };
    requireString("workspaceId", workspaceId);
    requireString("worktreePath", worktreePath);
    requireString("systemPrompt", systemPrompt);
    if (!providerConfig || typeof providerConfig !== "object") throw new Error("providerConfig required");
    await registry.create({ workspaceId, worktreePath, systemPrompt, providerConfig });
    return { ok: true };
  });

  d.register("agent.prompt", async (p) => {
    const { workspaceId, message } = p as { workspaceId: string; message: string };
    requireString("workspaceId", workspaceId);
    requireString("message", message);
    await registry.get(workspaceId).prompt(message);
    return { ok: true };
  });

  d.register("agent.continue", async (p) => {
    const { workspaceId } = p as { workspaceId: string };
    requireString("workspaceId", workspaceId);
    await registry.get(workspaceId).continueRun();
    return { ok: true };
  });

  d.register("agent.abort", async (p) => {
    const { workspaceId } = p as { workspaceId: string };
    requireString("workspaceId", workspaceId);
    registry.get(workspaceId).abort();
    return { ok: true };
  });

  d.register("agent.dispose", async (p) => {
    const { workspaceId } = p as { workspaceId: string };
    requireString("workspaceId", workspaceId);
    await registry.dispose(workspaceId);
    return { ok: true };
  });

  d.register("agent.list", () => ({ workspaceIds: registry.list() }));
}

function requireString(name: string, v: unknown): asserts v is string {
  if (typeof v !== "string" || v.length === 0) throw new Error(`${name} must be a non-empty string`);
}
```

## Task 19: WebSocket server

**Files:**
- Create: `sidecar/src/server.ts`

Bun's WebSocket server can bind to a Unix socket via `unix:` option (Bun >= 1.1). One server, broadcasts events to all connected clients (Phase 1 expects only one client — the Mac app — but the broadcast model is trivial and lets us reconnect cleanly).

- [ ] **Step 1: Implement `sidecar/src/server.ts`**

```ts
import { unlinkSync, existsSync } from "node:fs";
import { mkdir } from "node:fs/promises";
import { dirname } from "node:path";
import { Dispatcher } from "./dispatcher.js";
import { AgentRegistry } from "./agentRegistry.js";
import { registerMethods } from "./methods.js";
import { parseFrame, formatEvent } from "./rpc.js";
import { log } from "./logger.js";

export type ServerOptions = {
  socketPath: string;
  dataDir: string;
};

export async function startServer(opts: ServerOptions): Promise<{ stop: () => Promise<void> }> {
  await mkdir(dirname(opts.socketPath), { recursive: true });
  if (existsSync(opts.socketPath)) {
    try { unlinkSync(opts.socketPath); } catch {}
  }

  const clients = new Set<{ send: (d: string) => number }>();
  const sink = (workspaceId: string, ev: { type: string; [k: string]: unknown }) => {
    const frame = formatEvent(ev.type, { workspaceId, ...ev });
    for (const c of clients) {
      try { c.send(frame); } catch (e) { log.warn("send failed", { err: String(e) }); }
    }
  };

  const registry = new AgentRegistry(opts.dataDir, sink);
  const dispatcher = new Dispatcher();
  registerMethods(dispatcher, registry);

  const server = Bun.serve({
    unix: opts.socketPath,
    fetch(req, server) {
      if (server.upgrade(req)) return;
      return new Response("multiharness-sidecar; use a WebSocket client", { status: 426 });
    },
    websocket: {
      open(ws) {
        clients.add(ws);
        log.info("client connected", { count: clients.size });
      },
      close(ws) {
        clients.delete(ws);
        log.info("client disconnected", { count: clients.size });
      },
      async message(ws, raw) {
        const text = typeof raw === "string" ? raw : new TextDecoder().decode(raw);
        let req;
        try { req = parseFrame(text); }
        catch (e) {
          log.warn("bad frame", { err: String(e) });
          return;
        }
        const out = await dispatcher.dispatch(req.id, req.method, req.params);
        ws.send(out);
      },
    },
  });

  // signal readiness to the parent (SidecarManager watches stderr for this exact line)
  console.error("READY");
  log.info("listening", { socket: opts.socketPath });

  return {
    stop: async () => {
      await registry.disposeAll();
      server.stop();
      try { unlinkSync(opts.socketPath); } catch {}
    },
  };
}
```

> NOTE: Bun's `Bun.serve` `unix` option exists in current Bun versions. If the installed version doesn't support it, fall back to TCP on `127.0.0.1` with a 0-port allocation and write the chosen port to a `<dataDir>/sock/control.port` file. Verify against the installed Bun version when implementing.

## Task 21: End-to-end integration test

**Files:**
- Create: `sidecar/test/e2e.test.ts`

This is the most important test in the plan. It spins the real server up against a temp directory, opens a WebSocket to the Unix socket, sends `agent.create` + `agent.prompt`, and asserts that streamed events arrive — using a mock provider that satisfies the pi-ai `Model` interface.

Because the exact `Model` interface in pi-ai is non-trivial, we use the simplest possible approach: the test creates a workspace pointing at a `kind: "openai-compatible"` provider whose `baseUrl` points at a small fake HTTP server we run in-process, then assert events.

- [ ] **Step 1: Write the test**

```ts
import { describe, it, expect, beforeAll, afterAll } from "vitest";
import { mkdtempSync, mkdirSync, readFileSync, existsSync } from "node:fs";
import { tmpdir } from "node:os";
import { join } from "node:path";
import { startServer } from "../src/server.js";

let dataDir: string;
let worktree: string;
let socketPath: string;
let stopServer: (() => Promise<void>) | null = null;
let mockHttpServer: { stop: () => void; baseUrl: string } | null = null;

beforeAll(async () => {
  dataDir = mkdtempSync(join(tmpdir(), "mh-e2e-data-"));
  worktree = mkdtempSync(join(tmpdir(), "mh-e2e-wt-"));
  socketPath = join(dataDir, "sock", "control.sock");
  mkdirSync(join(dataDir, "sock"));

  // tiny mock OpenAI-compatible HTTP server that streams a single chat completion chunk
  const port = 19237 + Math.floor(Math.random() * 100);
  const server = Bun.serve({
    port,
    fetch(req) {
      if (req.url.endsWith("/v1/chat/completions")) {
        const stream = new ReadableStream({
          start(c) {
            const enc = new TextEncoder();
            c.enqueue(enc.encode(`data: ${JSON.stringify({ choices: [{ delta: { content: "hi" }, finish_reason: null }] })}\n\n`));
            c.enqueue(enc.encode(`data: ${JSON.stringify({ choices: [{ delta: {}, finish_reason: "stop" }] })}\n\n`));
            c.enqueue(enc.encode("data: [DONE]\n\n"));
            c.close();
          },
        });
        return new Response(stream, { headers: { "content-type": "text/event-stream" } });
      }
      if (req.url.endsWith("/v1/models")) {
        return Response.json({ data: [{ id: "mock-model" }] });
      }
      return new Response("not found", { status: 404 });
    },
  });
  mockHttpServer = {
    stop: () => server.stop(),
    baseUrl: `http://127.0.0.1:${port}/v1`,
  };

  const handle = await startServer({ socketPath, dataDir });
  stopServer = handle.stop;
});

afterAll(async () => {
  await stopServer?.();
  mockHttpServer?.stop();
});

describe("e2e: server + websocket + mock provider", () => {
  it("creates an agent and streams events from a prompt", async () => {
    const ws = new WebSocket(`ws+unix://${socketPath}`);
    await new Promise<void>((res, rej) => {
      ws.addEventListener("open", () => res(), { once: true });
      ws.addEventListener("error", () => rej(new Error("ws error")), { once: true });
    });

    const events: any[] = [];
    ws.addEventListener("message", (m) => {
      const obj = JSON.parse(typeof m.data === "string" ? m.data : "");
      events.push(obj);
    });

    function call(method: string, params: object): Promise<any> {
      return new Promise((res, rej) => {
        const id = crypto.randomUUID();
        const handler = (m: MessageEvent) => {
          const obj = JSON.parse(typeof m.data === "string" ? m.data : "");
          if (obj.id === id) {
            ws.removeEventListener("message", handler as any);
            obj.error ? rej(new Error(obj.error.message)) : res(obj.result);
          }
        };
        ws.addEventListener("message", handler as any);
        ws.send(JSON.stringify({ id, method, params }));
      });
    }

    await call("health.ping", {});

    await call("agent.create", {
      workspaceId: "w1",
      worktreePath: worktree,
      systemPrompt: "you are helpful",
      providerConfig: {
        kind: "openai-compatible",
        modelId: "mock-model",
        baseUrl: mockHttpServer!.baseUrl,
      },
    });

    await call("agent.prompt", { workspaceId: "w1", message: "say hi" });

    // Wait for agent_end
    await new Promise<void>((res, rej) => {
      const t = setTimeout(() => rej(new Error("timeout waiting for agent_end")), 5000);
      const i = setInterval(() => {
        if (events.some((e) => e.event === "agent_end")) {
          clearInterval(i); clearTimeout(t); res();
        }
      }, 50);
    });

    expect(events.find((e) => e.event === "agent_start")).toBeDefined();
    expect(events.find((e) => e.event === "message_end")).toBeDefined();

    // jsonl was written
    expect(existsSync(join(dataDir, "workspaces", "w1", "messages.jsonl"))).toBe(true);
    const log = readFileSync(join(dataDir, "workspaces", "w1", "messages.jsonl"), "utf8");
    expect(log).toMatch(/agent_end/);

    await call("agent.dispose", { workspaceId: "w1" });
    ws.close();
  }, 15_000);
});
```

Goal: assert event flow, not transport flavor.
